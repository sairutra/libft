13// This function gets the dst string, src string, and n bytes (full buffer)
14// it needs to concatenate at the end. the size it should copy is always
15// size - dst_len - 1. If size is not bigger than dst, it cannnot concat
16// therefore it returns the length of the string it wanted to create
17// if the copy size is bigger than src len than it automatically makes
18// the copy size equal to the size of src_len to not run of the string
19// then after checking if there is enough space in dst and the amount it
20// copies from src is the maximum it can copy the memory with memcpy and
21// null terminate the end.
22
23#include <libft.h>

25size_t ft_strlcat(char *dst, const char *src, size_t size)
26{
27 size_t dst_len;
28 size_t src_len;
29 size_t copy_size;
30
31 dst_len = ft_strlen(dst);
32 src_len = ft_strlen(src);
33 copy_size = size - dst_len - 1;
34 if (size <= dst_len)
35 return (size + src_len);
36 if (copy_size > src_len)
37 copy_size = src_len;
38 if (copy_size > 0)
39 {
40 ft_memcpy(dst + dst_len, src, copy_size);
41 dst[dst_len + copy_size] = '\0';
42 }
43 return (dst_len + src_len);
44}
